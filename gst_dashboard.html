        // --- Utility Functions (Currency/Percent Formatting) ---

        function formatValue(value, type) {

            if (type === 'currency') {

                // Use Indian formatting for crores/lakhs for better readability

                const crores = value / 10000000;

                if (crores >= 1) {

                    return `₹ ${crores.toFixed(2)} Cr`;

                } else if (value >= 100000) {

                    return `₹ ${(value / 100000).toFixed(2)} L`;

                }

                return `₹ ${Math.round(value).toLocaleString('en-IN')}`;

            } else if (type === 'percent') {

                return `${(value * 100).toFixed(2)} %`;

            }

            return value.toLocaleString('en-IN');

        }


        function calculateTrend(current, previous, type) {

            let trendClass;

            let trendValue;


            if (type === 'percent') {

                const difference = current - previous;

                trendValue = (difference * 100).toFixed(2);

                trendClass = difference >= 0 ? 'positive' : 'negative';

                return `<span class="${trendClass}">${difference >= 0 ? '▲' : '▼'} ${Math.abs(trendValue)} %pt (vs PY)</span>`;

            } else {

                const change = ((current - previous) / previous) * 100;

                trendValue = change.toFixed(2);

                trendClass = change >= 0 ? 'positive' : 'negative';

                return `<span class="${trendClass}">${change >= 0 ? '▲' : '▼'} ${Math.abs(trendValue)} % (vs PY)</span>`;

            }

        }


        function renderKPIs() {

            const container = document.getElementById('kpi-container');

            container.innerHTML = '';

            for (const title in financialData) {

                const data = financialData[title];

                const card = document.createElement('div');

                card.classList.add('kpi-card');

                const trendHtml = calculateTrend(data.fy24, data.fy23, data.format);


                card.innerHTML = `

                    <div class="kpi-title">${title}</div>

                    <div class="kpi-value">${formatValue(data.fy24, data.format)}</div>

                    <div class="kpi-trend">${trendHtml}</div>

                `;

                container.appendChild(card);

            }

        }
